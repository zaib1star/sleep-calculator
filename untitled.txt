<script>
function calculateSleep() {
  let time = document.getElementById("wakeTime").value;

  if (!time) {
    document.getElementById("result").innerHTML = "Please enter a time!";
    return;
  }

  let [hour, minute] = time.split(":");

  let wakeDate = new Date();
  wakeDate.setHours(hour);
  wakeDate.setMinutes(minute);

  let output = "<b>Best times to sleep:</b><br><br>";

  for (let i = 6; i >= 4; i--) {
    let sleepTime = new Date(wakeDate.getTime() - (i * 90 + 15) * 60000);
    output += "🕒 " + sleepTime.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) + "<br>";
  }

  document.getElementById("result").innerHTML = output;
}
</script>
